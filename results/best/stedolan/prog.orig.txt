#include <stdio.h>

// Global variables
long z, x, G;

// Main function
int main() {
    // Print the P1 header and image dimensions
    puts("P1\n80 80");
    
    // Read a hexadecimal number into G
    scanf("%10lx", &G);

    // Loop to generate and print the pattern
    for (z = 0; 3 - z / 2160; z++) {
        x = z % 81 / 8 - 5;
        if (5 > x) {
            putchar(!(16 >> (x ^ -(x < 1)) + 1 & G << 5 >> z / 648 * 5) ^ 49);
        } else {
            putchar(10);
        }
    }

    return 0;
}
