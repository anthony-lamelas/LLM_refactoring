```c
#include <stdio.h>

// Global variables
long z, x, G;

// Main function
int main() {
    // Print header for P1 format and read input
    puts("P1\n80 80");
    scanf("%10lx", &G);

    // Loop through each pixel
    for (z = 0; 3 - z / 2160; z++) {
        // Calculate the x position
        x = (z % 81) / 8 - 5;

        // Determine pixel output
        if (5 > x) {
            // Calculate and print the pattern character
            putchar(!(16 >> ((x ^ -(x < 1)) + 1) & G << 5 >> (z / 648) * 5) ^ 49);
        } else {
            // Print newline character
            putchar(10);
        }
    }

    return 0;
}
```
