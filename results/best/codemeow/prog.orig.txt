// Refactored code with clear comments
#include <stdio.h>
#include <unistd.h>
#include <sys/ioctl.h>
#include <sys/time.h>
#include <termios.h>

// Global variables
double constants[] = {
    2.5673396845218159e-289, 2.5673475286552589e-289, 5.0789948392480145e-321, 
    4.4759385888332619e-091, 2.2662500784516041e-085, 6.0135556610268572e-154, 
    6.0138114327429894e-154, 6.0134700169990685e-154, 6.0134703504179883e-154, 
    2.5663272049433059e+151, 2.5673651826636406e+151, 6.0134700183229171e-154, 
    5.3767695968532491e-299, 7.2911289729117876e-304, 6.1257806499948415e-062, 
    4.6555645993480313e+025, 6.5632629331317560e+299, 7.9070799242303223e-101, 
    3.2391739187041928e-317
};

int term_width, term_height, cursor_x, cursor_y, randomness_seed = 0;

void write_output(const char *buffer, long size) {
    write(STDOUT_FILENO, buffer, size);
}

unsigned char generate_random_bit() {
    randomness_seed ^= 8;
    return ++randomness_seed % 8;
}

int random_from_constants(int index) {
    return ((char *)&constants[index])[generate_random_bit()];
}

void write_two_chars(int index) {
    write_output(((char *)(constants + 14)) + index * 2, 2);
}

void write_three_chars(int index) {
    write_output(((char *)(constants + 16)) + index * 3, 3);
}

int extract_digit(int value, int position) {
    return value / (*((char *)(constants + 18) + position)) % 10 + 48;
}

void initialize_randomness() {
    char buffer[16];
    gettimeofday((struct timeval *)buffer, NULL);
    randomness_seed = buffer[0];
}

void get_terminal_size() {
    unsigned long size;
    ioctl(STDOUT_FILENO, TIOCGWINSZ, &size);
    term_width = size & 65535;
    term_height = size >> 16 & 65535;
    cursor_x = term_height / 2 - 12;
    cursor_y = term_width - 4;
}

void escape_sequence() {
    write_output("\033[", 2);
}

void set_bold() {
    escape_sequence();
    write_output("1m", 2);
}

void reset_attributes() {
    escape_sequence();
    write_output("0m", 2);
}

void switch_to_alternate_buffer() {
    escape_sequence();
    write_output("?1049h", 6);
    escape_sequence();
    write_output("2J", 2);
    escape_sequence();
    write_output("H", 1);
}

void move_cursor_to(int row, int col) {
    escape_sequence();
    write_output((char[]){extract_digit(row, 2), extract_digit(row, 1), extract_digit(row, 0)}, 3);
    write_output(";", 1);
    row = col;
    write_output((char[]){extract_digit(row, 2), extract_digit(row, 1), extract_digit(row, 0)}, 3);
    write_output("H", 1);
}

void set_attribute(int index) {
    escape_sequence();
    write_two_chars(index);
    write_output("m", 1);
}

void display_initial_graphics() {
    for (int i = 0; i < 18; i++) {
        move_cursor_to(cursor_y + i / 6, cursor_x + (i % 6) * 4);
        set_attribute(((char *)&constants)[randomness_seed++]);
        write_output(((char *)&constants) + (i << 2) + 24, 4);
    }
}

void sleep_for_a_short_time() {
    char buffer[16] = {0};
    buffer[11] = 1;
    nanosleep((const struct timespec *)buffer, NULL);
}

void toggle_attribute(int index) {
    int attribute_indexes[] = {3, 0, 1, 2, 0, 1};
    if (generate_random_bit() < 2) {
        set_bold();
        set_attribute(attribute_indexes[index - 2]);
    } else {
        set_attribute(attribute_indexes[index + 1]);
    }
}

void compute_movement(int type, int x, int y, int *new_x, int *new_y) {
    switch (type) {
        case 0:
            if (x <= 2 || y < 4) {
                *new_x = random_from_constants(12);
                *new_y = 0;
            } else if (x < 15) {
                *new_x = random_from_constants(12);
                *new_y = -(x % 2);
            } else {
                *new_x = random_from_constants(12);
                *new_y = random_from_constants(12) / 2;
            }
            break;
        case 3:
            *new_x = random_from_constants(12) * 1.5;
            *new_y = random_from_constants(13);
            break;
        case 4:
            *new_x = random_from_constants(12) / 2;
            *new_y = random_from_constants(12) / 2;
            break;
        default:
            *new_x = random_from_constants(13) * 1.5 * (1 - (type - 1) * 2);
            *new_y = random_from_constants(13) / 2;
            break;
    }
}

void draw_graphics(int x, int y, int type, int height) {
    int new_x, new_y, delay_counter = 5;
    while (height-- > 0) {
        int temp = 35 - height;
        compute_movement(type, temp, y, &new_x, &new_y);

        if (new_y > 0 && y > cursor_y - 2) new_y--;

        if ((height < 3) || (x < 3 && height < 10)) {
            draw_graphics(x, y, 4, height);
        } else if (x == 0 && height % 3 == 0) {
            if (delay_counter <= 0) {
                delay_counter = 3;
                draw_graphics(x, y, (randomness_seed++ % 2) + 1, height + 5);
            }
        }
        delay_counter--;
        x += new_x;
        y += new_y;
        if (x < 0) x = 0;
        if (y < 0) y = 0;
        move_cursor_to(y, x);
        toggle_attribute(type);
        write_three_chars(type);
        reset_attributes();
        sleep_for_a_short_time();
    }
}

int main() {
    get_terminal_size();
    initialize_randomness();
    switch_to_alternate_buffer();
    display_initial_graphics();
    draw_graphics(cursor_x + 10, cursor_y - 1, 0, 35);
    move_cursor_to(term_width, 0);
    return 0;
}
