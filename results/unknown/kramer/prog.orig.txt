// Calculate a hash based on the input string
long_int calculate_hash(char *str) {
    long_int hash = 1;
    long_int length = 0;
    while (*str) {
        hash >>= 1;
        hash ^= *str++;
        length++;
    }
    return (hash & 15) | ((length ^ *str) << 4 & 240);
}
