#include <time.h>
#include <stdio.h>

// Global variables
int a, b = 44, x, y, z;

// Main function
int main() {
    // Initialize variables if not already initialized
    if (!a) {
        a = 2551443;
        x = -b;
        y = 2 - b;
        z = ((time(0) - 592531) % a << 9) / a;
    } else {
        // Determine the character to output
        if (++x >= a) {
            x = -b;
            y += 4;
            putchar(10);  // Newline
        } else if (x < 0) {
            x = x * x + y * y < b * b ? a = 1 - x, -1 : x + 1;
            putchar(32);  // Space
        } else {
            // Choose character based on conditions
            int index = ((x < a * (~z & 255) >> 8) ^ z >> 8);
            putchar("#." [index]);
        }
        // Recursively call main until the condition is met
        if (y <= b) {
            main();
        }
    }
    return 0;
}
