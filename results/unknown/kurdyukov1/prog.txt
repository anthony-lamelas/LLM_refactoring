#include <time.h>
#include <stdio.h>

// Global variables
int a, b = 44, x, y, z;

// Main function
int main() {
    if (!a) {
        a = 2551443;
        x = -b;
        y = 2 - b;
        z = ((time(0) - 592531) % a << 9) / a;
    } else {
        putchar(
            ++x >= a ? (x = -b, y += 4, 10) :
            x < 0 ? (x = x * x + y * y < b * b ? a = 1 - x, -1 : x + 1, 32) :
            "#."[(x < a * (~z & 255) >> 8) ^ z >> 8]
        );
        if (y > b) {
            return 0;
        }
    }
    return main();
}
